13enum class PicoScenesLoggerAspect : uint16_t {
14 Misc = 0,
15 License,
16 Frontend,
17 Baseband,
18 SDR,
19 HAL,
20 Plugin,
21 Platform,
22};

24enum class PicoScenesLoggerLevel : uint16_t {
25 VeryVeryVerbose = 0,
26 VeryVerbose = 1,
27 Verbose = 2,
28 Debug = 3,
29 Detail = 4,
30 Info = 5,
31 Warn = 6,
32 Error = 7
33};

35class LoggingService {
36public:
37 static std::shared_ptr<LoggingService> getInstance();
38
39 void setLevelAll(PicoScenesLoggerLevel level);
40
41 void setLevelAll(const std::string &levelName);
42
43 void setLevel4Aspect(PicoScenesLoggerAspect aspect, PicoScenesLoggerLevel level);
44
45 void setLevel4Aspect(const std::string &aspectName, const std::string &levelName);
46
47 [[nodiscard]] PicoScenesLoggerLevel getLevel(PicoScenesLoggerAspect aspect) const noexcept;
48
49 [[nodiscard]] bool shouldLog(PicoScenesLoggerAspect aspect, PicoScenesLoggerLevel level) const;
50
51 template<typename... Args>
52 void print(PicoScenesLoggerAspect aspect, PicoScenesLoggerLevel level, Args &&... args) {
53 if (shouldLog(aspect, level)) {
54 auto message = print(std::forward<Args>(args)...);
55 log(aspect, level, message);
56 }
57 }
58
59 template<typename... Args>
60 void directPrint(PicoScenesLoggerAspect aspect, PicoScenesLoggerLevel level, Args &&... args) {
61 auto message = print(std::forward<Args>(args)...);
62 log(aspect, level, message);
63 }
64
65 template<typename... Args>
66 void printf(PicoScenesLoggerAspect aspect, PicoScenesLoggerLevel level, Args &&... args) {
67 if (shouldLog(aspect, level)) {
68 auto message = printf(std::forward<Args>(args)...);
69 log(aspect, level, message);
70 }
71 }
72
73 template<typename... Args>
74 void directPrintf(PicoScenesLoggerAspect aspect, PicoScenesLoggerLevel level, Args &&... args) {
75 auto message = printf(std::forward<Args>(args)...);
76 log(aspect, level, message);
77 }
78
79 void log(PicoScenesLoggerAspect aspect, PicoScenesLoggerLevel level, const std::string &message);
80
81 void log(PicoScenesLoggerAspect aspect, const std::exception &exception);
82
83private:
84 std::map<PicoScenesLoggerAspect, std::shared_ptr<void>> loggers;
85 std::shared_ptr<void> consoleSink{};
86 std::shared_ptr<void> rotarySink{};
87
88 LoggingService();
89
90 template<typename FMTTYPE, typename... Args>
91 static std::string print(const FMTTYPE& first, Args &&... args) {
92 auto &&message = fmt::format(fmt::runtime(first), std::forward<Args>(args)...);
93 return message;
94 }
95
96 template<typename... Args>
97 static std::string printf(Args &&... args) {
98 auto &&message = fmt::sprintf(std::forward<Args>(args)...);
99 return message;
100 }
101};
