100 template<typename Rep, typename Period>
101 struct timespec duration2Timespec(const std::chrono::duration<Rep, Period> &dur) {
102 auto sec_dur = std::chrono::duration_cast<std::chrono::seconds>(dur);
103 auto remain_dur = dur - sec_dur;
104 auto remain_nano_dur = std::chrono::duration_cast<std::chrono::nanoseconds>(remain_dur);
105 return timespec{
106 .tv_sec = sec_dur.count(),
107 .tv_nsec = remain_nano_dur.count(),
108 };
