215 template<typename ParseType>
216 ParseType readValueFromFilePath(const std::string& path) {
217 std::fstream fin(path, std::ios::in);
218 std::string line;
219 std::getline(fin, line);
220 fin.close();
221 if (line.empty()) {
222 throw std::invalid_argument(fmt::sprintf("The result of path \"%s\" is empty! \n", path));
223 }
224
225 if constexpr (std::is_same_v<ParseType, std::string>) {
226 return line;
227 } else {
228 double valueD = std::stod(line); // boost::lexical_cast cannot recognize '0x3' style
229 return static_cast<ParseType>(valueD);
230 }
231 }

241 template<typename ParseType>
242 int writeValue2FilePath(const std::string& path, const ParseType& value, const std::string& formatStr = "{}") {
243 auto valueString = fmt::format(fmt::runtime(formatStr), value);
244 std::fstream fout(path, std::ios::out);
245 try {
246 fout << valueString;
247 fout.close();
248 return 0;
249 } catch (const std::exception&) {
250 throw std::runtime_error(fmt::format("Cannot write value {} to path {}.", value, path));
251 }
252 }
