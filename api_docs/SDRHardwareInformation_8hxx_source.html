63 [[nodiscard]] const SDREthernetConnectionInterface* getPrimaryInterface() const {
64 if (connectionType != SDRConnectionType::Ethernet) return nullptr;
65 auto it = std::find_if(ethernetInterfaces.begin(), ethernetInterfaces.end(),
66 [](const auto& iface) { return iface.isPrimary; });
67 if (it != ethernetInterfaces.end()) {
68 return &(*it);
69 }
70 return ethernetInterfaces.empty() ? nullptr : &ethernetInterfaces.front();
71 }
72
77 [[nodiscard]] bool isDualConnection() const {
78 return ethernetInterfaces.size() > 1;
79 }

125 uint16_t deviceOrder{0};

135 // Add the other board to otherBoards

143 [[nodiscard]] size_t getTotalBoardCount() const {
144 return 1 + otherBoards.size(); // Current board + other connected boards
145 }
146
150 void buildInverseMap() {
151 // Clear existing map
152 inverseMap.clear();
153
154 // Add basic identifiers
155 if (!serial.empty()) inverseMap[serial] = "serial";
156 if (!name.empty()) inverseMap[name] = "name";
157 if (!resource.empty()) inverseMap[resource] = "resource";
158
159 // Add ethernet addresses if present
160 if (connectionProfile.connectionType == SDRConnectionProfile::SDRConnectionType::Ethernet) {
161 for (const auto& interface : connectionProfile.ethernetInterfaces) {
162 if (!interface.ipAddress.empty() && !inverseMap.contains(interface.ipAddress)) {
163 inverseMap[interface.ipAddress] = "addr";
164 }
165 }
166 }
167 }

651 std::string hardwareInformationSheet() {
652 std::ostringstream info;
653 const auto& board = getHardwareBoardInfo();
654
655 // Get maximum line width for border
656 size_t maxWidth = 100; // wider width for better readability
657
658 // Top border
659 info << "\n" << std::string(maxWidth, '*') << "\n";
660 // Print information for each board
661 const auto& boards = getAllHardwareBoards();
662 for (size_t i = 0; i < boards.size(); i++) {
663 const auto& board = boards[i];
664 info << "* Board " << i << ": " << board.name
665 << " [Serial: " << board.serial
666 << ", Model: " << board.model
667 << ", FPGA: " << board.fpgaImage
668 << ", Subtype: " << DeviceSubtype2String(board.deviceSubtype) << "]\n";
669
670 // Print connectivity information
671 const auto& conn = board.connectionProfile;
672 info << "* Connection: " << (conn.connectionType == SDRConnectionProfile::SDRConnectionType::Ethernet ?
673 "Ethernet" :
674 conn.connectionType == SDRConnectionProfile::SDRConnectionType::PCIe ? "PCIe" :
675 conn.connectionType == SDRConnectionProfile::SDRConnectionType::USB ? "USB" : "Unknown") << "\n";
676
677 if (conn.connectionType == SDRConnectionProfile::SDRConnectionType::Ethernet) {
678 for (const auto& eth : conn.ethernetInterfaces) {
679 info << "* Interface: " << eth.hostInterfaceName
680 << "=" << eth.ipAddress << " Speed=" << eth.linkSpeed / 1000 << "GbE MTU=" << eth.MTU << "\n";
681 }
682 }
683 }
684
685 // Section separator
686 info << "*" << std::string(maxWidth-2, '-') << "\n";
687 const auto& masterClockRates = getHardwareSupportedMasterClockRates();
688 info << "* Supported Master Clock Rates: [";
689 if (!masterClockRates.empty()) {
690 auto minRate = *std::min_element(masterClockRates.begin(), masterClockRates.end()) / 1e6;
691 auto maxRate = *std::max_element(masterClockRates.begin(), masterClockRates.end()) / 1e6;
692 info << minRate << " MHz, " << maxRate << " MHz";
693 }
694 info << "]\n";
695 info << "* Available Channel #: Tx (Transmit)=";
696 const auto& txChannels = getHardwareSupportedTxChannels();
697 if (txChannels.empty()) {
698 info << "N/A";
699 } else {
700 info << "[";
701 for (size_t chIndex = 0; chIndex < txChannels.size(); ++chIndex) {
702 info << (chIndex == 0 ? "" : ", ") << static_cast<int>(txChannels[chIndex]);
703 }
704 info << "]";
705 }
706 info << ", and Rx (Receive)=";
707 const auto& rxChannels = getHardwareSupportedRxChannels();
708 if (rxChannels.empty()) {
709 info << "N/A";
710 } else {
711 info << "[";
712 for (size_t chIndex = 0; chIndex < rxChannels.size(); ++chIndex) {
713 info << (chIndex == 0 ? "" : ", ") << static_cast<int>(rxChannels[chIndex]);
714 }
715 info << "]";
716 }
717 info << "\n";
718
719 // Section separator
720 info << "*" << std::string(maxWidth-2, '-') << "\n";
721
722 // Combined Channel-Frontend-Antenna Mapping section
723 info << "* <Motherboard -> Channel # (Tx/Rx) -> Tx/Rx FrontEnd -> Available Antennas> Mapping: \n";
724
725 // Process Tx channels
726 const auto& frontendProfiles = getHardwareFrontEndProfiles();
727 const auto& daughterboards = getHardwareDaughterboardProfiles();
728
729 // Helper function to get motherboard info for a frontend
730 auto getMboardInfo = [&](const std::string& frontendSerial) -> std::string {
731 for (const auto& db : daughterboards) {
732 for (const auto& fe : db.frontends) {
733 if (fe->serialNumber == frontendSerial) {
734 return fe->motherboardSerial + " (" +
735 DeviceSubtype2String(board.deviceSubtype) + ")";
736 }
737 }
738 }
739 return "N/A";
740 };
741
742 for (size_t i = 0; i < txChannels.size(); i++) {
743 uint8_t ch = txChannels[i];
744
745 // Start with motherboard info
746 info << "* ";
747
748 // Frontend info to get motherboard info
749 std::string frontendSerial;
750 for (const auto& profile : frontendProfiles) {
751 if (std::find(profile->txChannelNumber.begin(), profile->txChannelNumber.end(), ch) != profile->txChannelNumber.end()) {
752 frontendSerial = profile->serialNumber;
753 break;
754 }
755 }
756
757 // Print motherboard info
758 info << getMboardInfo(frontendSerial) << " -> ";
759
760 // Channel info
761 info << static_cast<int>(ch) << " (Tx) -> ";
762
763 // Frontend info
764 bool foundFrontend = false;
765 for (const auto& profile : frontendProfiles) {
766 if (std::find(profile->txChannelNumber.begin(), profile->txChannelNumber.end(), ch) != profile->txChannelNumber.end()) {
767 info << profile->serialNumber;
768 if (!profile->frontendHardwareModel.empty()) {
769 info << " (" << profile->frontendHardwareModel << ")";
770 }
771 foundFrontend = true;
772 break;
773 }
774 }
775 if (!foundFrontend) {
776 info << "N/A";
777 }
778
779 // Antenna info
780 info << " -> ";
781 const auto& supportedAnts = getHardwareSupportedTxAntennas(ch);
782 if (supportedAnts.empty()) {
783 info << "None";
784 } else {
785 for (size_t j = 0; j < supportedAnts.size(); j++) {
786 if (supportedAnts[j].empty()) {
787 info << "None";
788 } else {
789 info << supportedAnts[j];
790 }
791 if (j < supportedAnts.size() - 1) {
792 info << ", ";
793 }
794 }
795 }
796 info << "\n";
797 }
798
799 // Process Rx channels
800 for (size_t i = 0; i < rxChannels.size(); i++) {
801 uint8_t ch = rxChannels[i];
802
803 // Start with motherboard info
804 info << "* ";
805
806 // Frontend info to get motherboard info
807 std::string frontendSerial;
808 for (const auto& profile : frontendProfiles) {
809 if (std::find(profile->rxChannelNumber.begin(), profile->rxChannelNumber.end(), ch) != profile->rxChannelNumber.end()) {
810 frontendSerial = profile->serialNumber;
811 break;
812 }
813 }
814
815 // Print motherboard info
816 info << getMboardInfo(frontendSerial) << " -> ";
817
818 // Channel info
819 info << static_cast<int>(ch) << " (Rx) -> ";
820
821 // Frontend info
822 bool foundFrontend = false;
823 for (const auto& profile : frontendProfiles) {
824 if (std::find(profile->rxChannelNumber.begin(), profile->rxChannelNumber.end(), ch) != profile->rxChannelNumber.end()) {
825 info << profile->serialNumber;
826 if (!profile->frontendHardwareModel.empty()) {
827 info << " (" << profile->frontendHardwareModel << ")";
828 }
829 foundFrontend = true;
830 break;
831 }
832 }
833 if (!foundFrontend) {
834 info << "N/A";
835 }
836
837 // Antenna info
838 info << " -> ";
839 const auto& supportedAnts = getHardwareSupportedRxAntennas(ch);
840 if (supportedAnts.empty()) {
841 info << "None";
842 } else {
843 for (size_t j = 0; j < supportedAnts.size(); j++) {
844 if (supportedAnts[j].empty()) {
845 info << "None";
846 } else {
847 info << supportedAnts[j];
848 }
849 if (j < supportedAnts.size() - 1) {
850 info << ", ";
851 }
852 }
853 }
854 info << "\n";
855 }
856
857 // Section separator
858 info << "*" << std::string(maxWidth-2, '-') << "\n";
859
860 // Channel Capabilities section
861 info << "* Channel Capabilities:\n";
862
863 // Process Tx channels capabilities
864 for (size_t i = 0; i < txChannels.size(); i++) {
865 uint8_t ch = txChannels[i];
866 info << "* # " << static_cast<int>(ch) << " (Tx):\n";
867
868 // Gain range
869 info << "* Gain -> " << std::fixed << std::setprecision(1)
870 << getHardwareTxGainRange(ch).first << " to "
871 << getHardwareTxGainRange(ch).second << " dB\n";
872
873 // Frequency range
874 info << "* Carrier Frequency -> " << std::fixed << std::setprecision(1)
875 << getHardwareTxFrequencyRange(ch).first/1e6 << " to "
876 << getHardwareTxFrequencyRange(ch).second/1e6 << " MHz\n";
877
878 // Sampling rate range
879 info << "* Sampling Rate -> " << std::fixed << std::setprecision(2)
880 << getHardwareTxSamplingRateRange(ch).first/1e6 << " to "
881 << getHardwareTxSamplingRateRange(ch).second/1e6 << " MHz\n";
882 }
883
884 // Process Rx channels capabilities including AGC support
885 const auto& agcSupport = getHardwareAGCSupport();
886 for (size_t i = 0; i < rxChannels.size(); i++) {
887 uint8_t ch = rxChannels[i];
888 info << "* # " << static_cast<int>(ch) << " (Rx):\n";
889
890 // Gain range
891 info << "* Gain -> " << std::fixed << std::setprecision(1)
892 << getHardwareRxGainRange(ch).first << " to "
893 << getHardwareRxGainRange(ch).second << " dB\n";
894
895 // Frequency range
896 info << "* Carrier Frequency -> " << std::fixed << std::setprecision(1)
897 << getHardwareRxFrequencyRange(ch).first/1e6 << " to "
898 << getHardwareRxFrequencyRange(ch).second/1e6 << " MHz\n";
899
900 // Sampling rate range
901 info << "* Sampling Rate -> " << std::fixed << std::setprecision(2)
902 << getHardwareRxSamplingRateRange(ch).first/1e6 << " to "
903 << getHardwareRxSamplingRateRange(ch).second/1e6 << " MHz\n";
904
905 // AGC support
906 if (i < agcSupport.size() && agcSupport[i]) {
907 info << "* AGC -> Supported\n";
908 } else {
909 info << "* AGC -> Not Supported\n";
910 }
911 }
912
913 // Bottom border
914 info << std::string(maxWidth, '*') << "\n";
915
916 return info.str();
917 }
